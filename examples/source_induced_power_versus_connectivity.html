snr = 1.0
lambda2 = 1.0 / snr ** 2
psds_alpha = []

for lab in aparc:
    if lab.name.find('unknown') != -1:
        continue
    # compute the source space power and the inter-trial coherence
    from mne.minimum_norm import compute_source_psd_epochs

    stcs = compute_source_psd_epochs(epochs, inv, lambda2=lambda2, method='MNE',
                              fmin=8.0, fmax=13.0, pick_ori=None, label=lab,
                              nave=1, pca=True, inv_split=None, bandwidth=4.0,
                              adaptive=False, low_bias=True, return_generator=False,
                              n_jobs=4, prepared=False, verbose='ERROR')

    psds_alpha.append(np.mean([stc.data.mean() for stc in stcs]))  # average over freqs, sources

psds_alpha = np.array(psds_alpha)
